  var w = $("#chart_parent").width() * 0.99, h = w * 0.6;
  var is_small = false;

  if (Foundation.MediaQuery.atLeast('xlarge'))
  {
  }
  else if (Foundation.MediaQuery.atLeast('large'))
  {
  }
  else if (Foundation.MediaQuery.atLeast('medium'))
  {
  }
  else
  {
    is_small = true;
    h = w * 1.2;
  }

  w = parseInt(w);
  h = parseInt(h);

  var new_url = base_url.replace("XXXX", w);
  new_url = new_url.replace("YYYY", h);

      g_chart_controls.loaderlay.hide();
